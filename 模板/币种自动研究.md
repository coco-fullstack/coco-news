<%*
// 输入币种名称
const query = await tp.system.prompt("输入币种名称（如 kite, bitcoin, ethereum）");
if (!query) return;

try {
  // 搜索币种
  const searchUrl = `https://api.coingecko.com/api/v3/search?query=${encodeURIComponent(query)}`;
  const searchRes = await requestUrl({url: searchUrl});
  const coins = searchRes.json.coins;

  if (!coins || coins.length === 0) {
    new Notice("未找到该币种，请检查名称");
    return;
  }

  const coin = coins[0];
  const coinId = coin.id;
  const coinSymbol = coin.symbol.toUpperCase();
  const coinName = coin.name;

  // 获取详细数据
  const dataUrl = `https://api.coingecko.com/api/v3/coins/${coinId}?localization=false&tickers=false&community_data=false&developer_data=false`;
  const dataRes = await requestUrl({url: dataUrl});
  const d = dataRes.json;

  // 解析数据
  const price = d.market_data?.current_price?.usd != null ? d.market_data.current_price.usd : "N/A";
  const mcap = d.market_data?.market_cap?.usd ? "$" + (d.market_data.market_cap.usd / 1e6).toFixed(2) + "M" : "待更新";
  const fdv = d.market_data?.fully_diluted_valuation?.usd ? "$" + (d.market_data.fully_diluted_valuation.usd / 1e6).toFixed(2) + "M" : "待更新";
  const change24h = d.market_data?.price_change_percentage_24h != null ? d.market_data.price_change_percentage_24h.toFixed(2) + "%" : "N/A";
  const change7d = d.market_data?.price_change_percentage_7d != null ? d.market_data.price_change_percentage_7d.toFixed(2) + "%" : "N/A";
  const vol = d.market_data?.total_volume?.usd ? "$" + (d.market_data.total_volume.usd / 1e6).toFixed(2) + "M" : "N/A";
  const desc = d.description?.en ? d.description.en.split('. ').slice(0, 2).join('. ') : "待补充";
  const website = d.links?.homepage?.[0] || "";
  const twitter = d.links?.twitter_screen_name ? "https://x.com/" + d.links.twitter_screen_name : "";
  const categories = d.categories?.filter(c => c)?.join(", ") || "待分类";
  const rank = d.market_cap_rank || "N/A";

  // 重命名文件
  await tp.file.rename(coinName);

  // 输出模板
  tR += `---
title: ${coinName}
date: ${tp.date.now("YYYY-MM-DD")}
tags:
  - research/token
  - token/${coinSymbol}
related: []
status: active
---

## 基本信息

- **全称**：${coinName}
- **代币**：${coinSymbol}
- **排名**：#${rank}
- **赛道**：${categories}
- **官网**：${website}
- **Twitter**：${twitter}

## 项目简介

${desc}

## 实时数据（${tp.date.now("YYYY-MM-DD")} 自动获取）

| 指标 | 数值 |
|------|------|
| 价格 | $${price} |
| 市值 | ${mcap} |
| FDV | ${fdv} |
| 24h 成交量 | ${vol} |
| 24h 涨跌 | ${change24h} |
| 7d 涨跌 | ${change7d} |

## 核心逻辑

- 解决什么问题？
- 竞争优势是什么？
- 代币经济模型：

## 风险

- [ ] 竞品对比

## 我的判断

- **评级**：关注 / 建仓 / 观望 / 回避
- **理由**：
- **仓位计划**：
`;

} catch (e) {
  new Notice("获取数据失败：" + e.message);
}
